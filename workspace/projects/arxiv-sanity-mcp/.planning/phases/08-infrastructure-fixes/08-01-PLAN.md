```python
"""Import isolation tests for content package.

Ensures importing content models does not eagerly load heavy dependencies.
"""
import importlib
import sys


class TestContentImportIsolation:
    def test_content_models_no_httpx(self):
        """SC-4: Importing content.models must not pull in httpx."""
        # Remove cached modules to test fresh import chain
        mods_to_remove = [k for k in sys.modules if k.startswith("arxiv_mcp.content")]
        for mod in mods_to_remove:
            del sys.modules[mod]

        # Import just the models submodule
        importlib.import_module("arxiv_mcp.content.models")

        # httpx should not have been loaded as a side effect
        # (it may already be loaded from other tests, so we check
        # that content.html_fetcher was NOT loaded)
        assert "arxiv_mcp.content.html_fetcher" not in sys.modules, (
            "Importing content.models should not trigger html_fetcher import"
        )
```

Note: Since test ordering may mean httpx is already in sys.modules from other tests, the test checks for the html_fetcher module itself rather than httpx. This is the correct proxy -- if html_fetcher is not loaded, httpx was not pulled by the content package.
